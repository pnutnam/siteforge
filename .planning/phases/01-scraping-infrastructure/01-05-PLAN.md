Add POST /api/scrape/start endpoint to src/dashboard/routes.ts.

Read the existing routes.ts file first to understand the current structure.

The new endpoint should:

1. POST /api/scrape/start
2. Read tenantId from x-tenant-id header (return 401 if missing)
3. Validate required fields in body: businessId, name, address
4. Call startBusinessScrape({ businessId, tenantId, name, address, url })
5. Return 202 Accepted with { jobId, status: 'accepted' } on success
6. Return 400 with error message on validation failure

Example implementation:
```typescript
router.post('/scrape/start', async (req: Request, res: Response) => {
  const tenantId = (req as any).tenantId;
  if (!tenantId) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const { businessId, name, address, url } = req.body;
  if (!businessId || !name || !address) {
    return res.status(400).json({ error: 'Missing required fields: businessId, name, address' });
  }

  try {
    const flow = await startBusinessScrape({
      businessId,
      tenantId,
      name,
      address,
      url: url || '',
    });
    res.status(202).json({ status: 'accepted', flowId: (flow as any).id });
  } catch (error) {
    res.status(500).json({ error: 'Failed to start scrape', details: error instanceof Error ? error.message : String(error) });
  }
});
```

Import startBusinessScrape at the top of routes.ts:
```typescript
import { startBusinessScrape } from '../jobs/producers';
```
